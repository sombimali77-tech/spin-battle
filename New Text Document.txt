                 background: conic-gradient(red 0deg 180deg, yellow 180deg 360deg); }

    <img id="winnerImg" src="">
    <audio id="spinAudio" src="spin.mp3"></audio>
    <audio id="winAudio" src="win.mp3"></audio>

    <script>
        let red = 0, yellow = 0, currentDeg = 0;
        const wheel = document.getElementById('wheel');

        function spin() {
            document.getElementById('spinAudio').play();
            let direction = Math.random() > 0.5 ? 1 : -1;
            let targetDeg = currentDeg + (direction * (3600 + Math.random() * 360));
            
            // Force the animation to last exactly 30 seconds
            wheel.animate([
                { transform: `rotate(${currentDeg}deg)` },
                { transform: `rotate(${targetDeg}deg)` }
            ], {
                duration: 30000, 
                easing: 'cubic-bezier(0.1, 0.7, 0.1, 1)',
                fill: 'forwards'
            });

            currentDeg = targetDeg;

            setTimeout(() => {
                document.getElementById('winAudio').play();
                let finalDeg = ((targetDeg % 360) + 360) % 360;
                if (direction === -1) finalDeg = 360 - finalDeg;

                let img = document.getElementById('winnerImg');
                if(finalDeg < 180) { red++; document.getElementById('redCount').innerText = red; img.src = 'ronaldo.jpg'; }
                else { yellow++; document.getElementById('yellowCount').innerText = yellow; img.src = 'messi.jpg'; }
                
                img.style.display = 'block';
                setTimeout(() => { img.style.display = 'none'; spin(); }, 10000);
            }, 30000);
        }
        document.body.onclick = () => { document.body.onclick = null; spin(); };
    </script>
</body>
